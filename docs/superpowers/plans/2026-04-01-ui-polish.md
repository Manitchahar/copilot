```js
const ACTION_PATTERNS = [
  "bash",
  "write",
  "edit",
  "create",
  "delete",
  "run",
  "git",
  "execute",
  "install",
  "build",

const RESEARCH_PATTERNS = [
  "read",
  "grep",
  "search",
  "ls",
  "glob",
  "find",
  "view",
  "list",
  "cat",
  "head",
  "tail",
];

/**
 * Classify a tool name into a visibility tier.
 * - "action"   -> prominent card (bash, write, edit, git, etc.)
 * - "research" -> collapsed summary (read, grep, search, ls, etc.)
 * Action patterns are checked first so tools like "edit_and_read" are Action.
 * Unrecognised tools default to "action".
 */
export function classifyToolEvent(toolName) {
  if (!toolName) return "action";
  const lower = toolName.toLowerCase();
  if (ACTION_PATTERNS.some((p) => lower.includes(p))) return "action";
  if (RESEARCH_PATTERNS.some((p) => lower.includes(p))) return "research";
  return "action";
}
```

- [ ] **Step 5: Create `src/hooks/useRelativeTime.js`**

```js
import { useEffect, useState } from "react";

/**
 * Format a timestamp as relative time ("just now", "2m ago", "1h ago").
 */
export function formatRelativeTime(timestamp) {
  if (!timestamp) return "";
  const seconds = Math.floor((Date.now() - timestamp) / 1000);
  if (seconds < 5) return "just now";
  if (seconds < 60) return `${seconds}s ago`;
  const minutes = Math.floor(seconds / 60);
  if (minutes < 60) return `${minutes}m ago`;
  const hours = Math.floor(minutes / 60);
  if (hours < 24) return `${hours}h ago`;
  const days = Math.floor(hours / 24);
  return `${days}d ago`;
}

/**
 * Hook that returns a relative-time string and re-renders every 60s.
 * Shows full date on hover via the title attribute of the parent element.
 */
export function useRelativeTime(timestamp) {
  const [, setTick] = useState(0);

  useEffect(() => {
    if (!timestamp) return;
    const interval = setInterval(() => setTick((t) => t + 1), 60_000);
    return () => clearInterval(interval);
  }, [timestamp]);

  return formatRelativeTime(timestamp);
}
```

- [ ] **Step 1: Create `src/hooks/useSmartScroll.js`**

Ported from `src/pages/ChatPrototype.jsx` lines 22-68. Uses IntersectionObserver on a sentinel div to detect if the user is near the bottom of the scroll container. Auto-scrolls on content changes only if near bottom; shows a "new messages" pill otherwise.

```js
import { useCallback, useEffect, useRef, useState } from "react";

/**
 * Smart scroll hook that auto-scrolls only when the user is near the bottom.
 * Returns refs for the container and sentinel elements, a showPill flag,
 * and a scrollToBottom function.
 *
 * @param {Array} deps - dependency array that triggers scroll checks
 */
export function useSmartScroll(deps) {
  const containerRef = useRef(null);
  const sentinelRef = useRef(null);
  const isNearBottom = useRef(true);
  const [showPill, setShowPill] = useState(false);

  useEffect(() => {
    const sentinel = sentinelRef.current;
    if (!sentinel) return;

    const observer = new IntersectionObserver(
      ([entry]) => {
        isNearBottom.current = entry.isIntersecting;
        if (entry.isIntersecting) setShowPill(false);
      },
      { root: containerRef.current, threshold: 0.1 }
    );

    observer.observe(sentinel);
    return () => observer.disconnect();
  }, []);

  useEffect(() => {
    if (isNearBottom.current && containerRef.current) {
      containerRef.current.scrollTo({
        top: containerRef.current.scrollHeight,
        behavior: "smooth",
      });
    } else if (!isNearBottom.current) {
      setShowPill(true);
    }
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, deps);

  const scrollToBottom = useCallback(() => {
    if (containerRef.current) {
      containerRef.current.scrollTo({
        top: containerRef.current.scrollHeight,
        behavior: "smooth",
      });
      setShowPill(false);
    }
  }, []);

  return { containerRef, sentinelRef, showPill, scrollToBottom };
}
```

```jsx
import { useState } from "react";

export default function CodeBlock({ children, className }) {
  const [copied, setCopied] = useState(false);
  const language = className?.replace("language-", "") || "";
  const code = String(children).replace(/\n$/, "");

  const handleCopy = async () => {
    try {
      await navigator.clipboard.writeText(code);
      setCopied(true);
      setTimeout(() => setCopied(false), 2000);
    } catch {
      /* clipboard not available in insecure context */
    }
  };

  return (
    <div className="group relative my-3 overflow-hidden rounded-lg bg-[#1a1c1a]">
      <div className="flex items-center justify-between px-4 py-2">
        <span className="font-mono text-[11px] uppercase tracking-wide text-[#faf9f6]/50">
          {language || "code"}
        </span>
        <button
          onClick={handleCopy}
          className="flex items-center gap-1 rounded px-2 py-1 font-label text-[11px] text-[#faf9f6]/50 transition-colors hover:text-[#faf9f6]"
        >
          <span className="material-symbols-outlined text-sm">
            {copied ? "check" : "content_copy"}
          </span>
          {copied ? "Copied" : "Copy"}
        </button>
      </div>
      <pre className="overflow-x-auto px-4 pb-4 font-mono text-[13px] leading-relaxed text-[#faf9f6]">
        <code className={className}>{children}</code>
      </pre>
    </div>
  );
}
```

- [ ] **Step 1: Create `src/hooks/useAutoResize.js`**

Returns a ref that auto-resizes a textarea to fit its content, up to a maximum height (default 200px, approximately 6 lines of text).

```js
import { useEffect, useRef } from "react";

/**
 * Hook that returns a ref for a textarea element.
 * The textarea auto-resizes its height to fit content, up to maxHeight.
 *
 * @param {string} value - current textarea value (triggers resize on change)
 * @param {number} maxHeight - maximum height in pixels (default 200)
 */
export function useAutoResize(value, maxHeight = 200) {
  const ref = useRef(null);

  useEffect(() => {
    const el = ref.current;
    if (!el) return;
    el.style.height = "auto";
    el.style.height = Math.min(el.scrollHeight, maxHeight) + "px";
  }, [value, maxHeight]);

  return ref;
}
```

```jsx
import { useSmartScroll } from "../../hooks/useSmartScroll";
import MessageGroup from "./MessageGroup";
import TypingIndicator from "./TypingIndicator";
import ScrollPill from "./ScrollPill";

function groupMessages(messages) {
  const groups = [];
  let currentGroup = [];

  for (const msg of messages) {
    if (currentGroup.length > 0 && currentGroup[0].role !== msg.role) {
      groups.push(currentGroup);
      currentGroup = [];
    }
    currentGroup.push(msg);
  }
  if (currentGroup.length > 0) groups.push(currentGroup);

  return groups;
}

export default function MessageList({

  streaming,
  header,
  footer,

    useSmartScroll([messages, busy]);
```
